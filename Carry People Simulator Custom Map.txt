--Services
local tweenservice = game:GetService("TweenService")

local player = game:GetService("Players").LocalPlayer
player.Character.HumanoidRootPart.CFrame = CFrame.new(0,10010,0)
player.Character.HumanoidRootPart.Anchored = true

pcall(function()
    if workspace.CurrentCamera.CustomMap then
        workspace.CurrentCamera.CustomMap:Destroy()
    end
end)

wait()

function build(part,buildtime,sizex,sizey,sizez)
    local tweena = tweenservice:Create(part, TweenInfo.new(buildtime), {Size = Vector3.new(sizex,sizey,sizez)})
    tweena:Play()
    tweena.Completed:Wait()
end

local baseplate = Instance.new("Part")
baseplate.Name = "CustomMap"
baseplate.Parent = workspace.CurrentCamera
baseplate.CFrame = CFrame.new(0,10000,0)
baseplate.Size = Vector3.new(0,7,0)
baseplate.Anchored = true

local buildings = Instance.new("Folder")
buildings.Name = "Buildings"
buildings.Parent = baseplate

build(baseplate,0.7,65,7,65)

player.Character.HumanoidRootPart.Anchored = false
